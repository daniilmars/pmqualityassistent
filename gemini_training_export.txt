--- File: .github/workflows/deploy.yml ---

name: Deploy Website to Netlify/GitHub Pages

on:
  push:
    branches:
      - main

permissions:
  contents: read
  pages: write
  id-token: write

jobs:
  deploy:
    environment:
      name: github-pages
      url: ${{ steps.deployment.outputs.page_url }}
    
    runs-on: ubuntu-latest
    
    steps:
      - name: Checkout code
        uses: actions/checkout@v4

      - name: Setup Pages
        uses: actions/configure-pages@v4

      - name: Upload artifact
        uses: actions/upload-pages-artifact@v3
        with:
          path: '.'

      - name: Deploy to GitHub Pages
        id: deployment
        uses: actions/deploy-pages@v4

document.addEventListener('DOMContentLoaded', () => {

    // --- Code für sanftes Scrollen ---
    document.querySelectorAll('a[href^="#"]').forEach(anchor => {
        anchor.addEventListener('click', function (e) {
            e.preventDefault();
            const targetId = this.getAttribute('href');
            document.querySelector(targetId).scrollIntoView({
                behavior: 'smooth'
            });
        });
    });

    // --- Code für Fade-In Animationen beim Scrollen ---
    const observer = new IntersectionObserver((entries) => {
        entries.forEach(entry => {
            if (entry.isIntersecting) {
                entry.target.classList.add('is-visible');
            }
        });
    }, 
    {
        threshold: 0.1
    });

    // Weisen Sie den Observer allen Sektionen und Karten zu
    document.querySelectorAll('section, .card, .pilot-box').forEach(element => {
        element.classList.add('fade-in-section');
        observer.observe(element);
    });

    // --- (Corrected) Code für Timeline ---
    const processSection = document.querySelector('.process-section');
    if (processSection) {
        
        // 1. ANIMATION OBSERVER
        const timelineAnimationObserver = new IntersectionObserver((entries) => {
            entries.forEach(entry => {
                if (entry.isIntersecting) {
                    const paths = document.querySelectorAll('.timeline-svg .timeline-path');
                    paths.forEach(path => {
                        path.style.transition = 'stroke-dashoffset 4s ease-out';
                        path.style.strokeDashoffset = '0';
                    });
                    timelineAnimationObserver.unobserve(entry.target);
                }
            });
        }, {
            threshold: 0.5 
        });
        timelineAnimationObserver.observe(processSection);

        // 2. TOOLTIP INTERACTIVITY (CORRECTED & FINAL)
        const tooltip = document.getElementById('timeline-tooltip');
        const milestoneCircles = document.querySelectorAll('.milestone-circle');

        milestoneCircles.forEach(circle => {
            circle.addEventListener('mouseenter', () => {
                const tooltipText = circle.parentNode.getAttribute('data-tooltip');
                tooltip.innerHTML = tooltipText;

                const circleRect = circle.getBoundingClientRect();
                const tooltipWidth = 220; // Width from CSS

                let top = circleRect.top + window.scrollY - 45; // Adjusted offset
                // THE FIX IS HERE: Removed "+ window.scrollY" from the 'left' calculation
                let left = circleRect.left + (circleRect.width / 2) - (tooltipWidth / 2);

                if (left < 10) left = 10;

                tooltip.style.top = `${top}px`;
                tooltip.style.left = `${left}px`;
                tooltip.style.display = 'block';
                tooltip.style.opacity = 1;
            });

            circle.addEventListener('mouseleave', () => {
                tooltip.style.opacity = 0;
                // Hide the element after the fade-out transition (200ms)
                setTimeout(() => {
                    tooltip.style.display = 'none';
                }, 200);
            });
        });
    }
});
